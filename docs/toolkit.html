10 slide layout types
coverTitle slide — always first
{ "type": "cover" }
// Uses top-level title, subtitle, presenter, date
agendaNumbered agenda list
{ "type": "agenda",
  "title": "Today's Agenda",
  "items": ["Section 1", "Section 2"] }
sectionFull-slide section divider
{ "type": "section",
  "number": "01", "title": "Section Title",
  "subtitle": "Optional sub" }
stat_row2–4 big metric boxes
{ "type": "stat_row", "title": "At a Glance",
  "stats": [{ "value": "$2.6M",
    "label": "ARR", "color": "green" }] }
bulletsTitle + bullet list — most common
{ "type": "bullets",
  "title": "Slide Title",
  "bullets": ["Point one", "Point two"] }
two_colSide-by-side columns
{ "type": "two_col", "title": "Comparison",
  "left":  { "heading": "A", "bullets": [...] },
  "right": { "heading": "B", "bullets": [...] } }
tableData table — RAG colours auto-applied
{ "type": "table",
  "headers": ["KPI", "Target", "Status"],
  "rows": [["NRR", "+10%", "🟢 Ahead"]] }
cards2–4 feature cards with icons
{ "type": "cards",
  "cards": [{ "icon": "🏆",
    "title": "Expert", "body": "..." }] }
timelineHorizontal phase roadmap
{ "type": "timeline",
  "steps": [{ "phase": "Q1",
    "title": "Foundation", "status": "done" }] }
closingThank you / final slide
{ "type": "closing",
  "title": "Thank You",
  "contact": "name@uipath.com" }
